(defun make-project-dir-from-heading ()
  (interactive)
  (save-restriction
    (org-narrow-to-subtree)
    (let ((heading-title (nth 4 (org-heading-components))))
      (make-directory heading-title t)
      (goto-char (point-max))
      (insert (format "[[file:%s]]" heading-title)))))
