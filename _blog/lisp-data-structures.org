(setq data '(("p1" . (:points 10 
				:objectives '(1 2)
				:abet '(a k)))
	     ("p2"   . (:points 20 :abet '(j)))
	     ("p3"   . (:abet 'b))))


(defun get (problem property)
  (interactive)
  (plist-get (cdr (assoc problem data)) property))

(mapcar (lambda (x) (get (car x) :abet)) data)
